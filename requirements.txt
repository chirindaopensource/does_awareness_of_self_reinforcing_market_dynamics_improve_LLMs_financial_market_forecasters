# Core n-dimensional array library.
# Used for:
#   • float64 price (Pₜ) and EPS (EPSₜ) array storage (288-element vectors)
#   • Vectorized normalization arithmetic:
#       N1:  P'_k  = 100 * price_array[t0-1 : t0+W-1] / P0
#       N2:  EPS'_k = 100 * eps_array[t0-1 : t0+W-1]  / P0
#   • numpy.round(array, decimals=2) for 2 d.p. serialization rounding
#   • numpy.mean(r_port) and numpy.std(r_port, ddof=1) for Sharpe computation
#       Note: ddof=1 is the Bessel-corrected sample estimator; ddof=0 is WRONG
#   • numpy.sqrt(12) annualization factor (= 3.4641016151377544)
#   • numpy.concatenate([dotcom_returns, gfc_returns]) for combined 144-obs series
#   • numpy.allclose(P_prime/EPS_prime, P_raw/EPS_raw, rtol=0, atol=1e-10)
#       for P/E ratio preservation invariant verification
#   • numpy.float64 dtype enforcement on price and EPS arrays
# Minimum version rationale:
#   1.24.0 introduces stable structured dtype handling and eliminates
#   known precision regressions in vectorized reduction operations.
# Upper bound: <3.0.0 guards against NumPy 3.x breaking changes to
#   ufunc signatures and array protocol.
numpy>=1.24.0,<3.0.0

# Excel engine for reading legacy .xls format Shiller workbooks.

# Upper bound: <3.0.0 is forward protection.

# OpenAI Python SDK — for GPT-5 (temperature=1.0; stochastic; irreproducible).
# API call pattern:
#   client = openai.OpenAI(api_key=OPENAI_API_KEY)
#   response = client.chat.completions.create(
#       model="gpt-5",
#       temperature=1.0,
#       max_tokens=512,
#       messages=[
#           {"role": "system", "content": system_prompt},
#           {"role": "user",   "content": time_series_table}
#       ]
#   )
# CRITICAL REPRODUCIBILITY WARNING:
#   GPT-5 at temperature=1.0 with no seed parameter produces a fresh
#   stochastic draw on every API call. The paper's GPT-5 results are
#   single-run point estimates with no reported variance. Exact numerical
#   reproduction of GPT-5 results is categorically impossible under this
#   specification (severity: FATAL per STUDY_CONFIG).
# INCOMPATIBILITY WARNING:
#   The pre-1.0.0 openai SDK used module-level functions (openai.ChatCompletion)
#   which are structurally incompatible with the 1.x client-constructor pattern.
#   Installing openai<1.0.0 will cause AttributeError at client instantiation.
# Minimum version rationale:
#   1.40.0 ensures stability of the chat.completions interface and captures
#   all model registry entries for gpt-5 (released 2025-08-07).
# Upper bound: <2.0.0 guards against a potential 2.x rewrite of the SDK API.
openai>=1.40.0,<2.0.0

# Anthropic Python SDK — for Claude Sonnet 4.6 (temperature=0; deterministic).
# API call pattern (NOTE: system prompt is a TOP-LEVEL parameter, NOT in messages):
#   client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)
#   response = client.messages.create(
#       model="claude-sonnet-4-6",
#       max_tokens=512,
#       temperature=0,
#       system=system_prompt,          # TOP-LEVEL — NOT inside messages list
#       messages=[
#           {"role": "user", "content": time_series_table}
#       ]
#   )
# STRUCTURAL TRAP: Placing the system prompt inside messages as
#   {"role": "system", "content": ...} will raise an Anthropic API
#   ValidationError. The system= parameter placement is mandatory.
# SDK >=0.20.0 automatically injects the required header:
#   anthropic-version: 2023-06-01
# Model ID: "claude-sonnet-4-6" (dateless canonical ID for the 4.6 generation;
#   no date suffix required per Anthropic documentation).
# Minimum version rationale:
#   0.25.0 stabilizes the messages.create() interface and the system=
#   top-level parameter pattern used by this pipeline.
# Upper bound: <1.0.0 guards against post-stabilization breaking rewrites.
anthropic>=0.25.0,<1.0.0

# Google GenerativeAI Python SDK — for Gemini 3 Pro (temperature=0).
# pip package: google-generativeai   →   import google.generativeai
# (Note: "google-generativeai" ≠ "google-genai"; these are distinct packages
#  with incompatible import namespaces. The correct package for
#  `import google.generativeai` is `google-generativeai`.)
# API call pattern:
#   import google.generativeai as genai
#   client = genai.GenerativeModel(model_name="gemini-3-pro-preview")
#   response = client.generate_content(
#       contents=[{"role": "user", "parts": [{"text": time_series_table}]}],
#       generation_config=genai.types.GenerationConfig(
#           temperature=0,
#           max_output_tokens=512   # NOTE: "max_output_tokens", NOT "max_tokens"
#       ),
#       system_instruction=system_prompt   # Google's name for system param
#   )
# PROVIDER-SPECIFIC TRAPS:
#   • Parameter name is max_output_tokens (NOT max_tokens as in OpenAI/Anthropic)
#   • System prompt key is system_instruction (NOT system or role:system)
# DEPRECATION WARNING:
#   "gemini-3-pro-preview" was deprecated on 2026-03-09 and now redirects to
#   "gemini-3.1-pro-preview". Post-deprecation replication must substitute
#   the successor model ID. This substitution is itself a source of model
#   version drift (severity: HIGH per STUDY_CONFIG).
# Minimum version rationale:
#   0.8.0 provides the generate_content() interface with system_instruction
#   support required by this pipeline.
# Upper bound: <2.0.0 is forward protection.
google-generativeai>=0.8.0,<2.0.0
